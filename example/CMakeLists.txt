cmake_minimum_required(VERSION 2.8.12)

if (POLICY CMP0054)
    cmake_policy(SET CMP0054 NEW)
endif ()

set(EXAMPLES
        create_bencode
        document
        file
        parse
        sax
        stream_wrapper
        struct)

include_directories(".")

endforeach ()

add_custom_target(examples ALL DEPENDS ${EXAMPLES})
